<script>
    let db = { players: [], courses: [], rounds: [] };
    const KEY = "CADDY_V69_STABLE_FINAL";

    // --- NAVIGATION LOGIC ---
    function initNav() {
        const btns = document.querySelectorAll('.tab-btn');
        btns.forEach(btn => {
            btn.addEventListener('click', () => {
                const targetId = btn.getAttribute('data-target');
                
                // Remove active from all
                document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
                document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
                
                // Add to selected
                document.getElementById(targetId).classList.add('active');
                btn.classList.add('active');
                
                if(targetId === 'pg-stats') renderStats();
                if(targetId === 'pg-rounds') renderRounds();
                window.scrollTo(0,0);
            });
        });
    }

    // --- CORE LOGIC ---
    function sync() {
        localStorage.setItem(KEY, JSON.stringify(db));
        updateDropdowns();
        renderPlayerList();
        renderCourseList();
    }

    function buildScorecard(data = null) {
        const cid = document.getElementById('pickCourse').value;
        const c = db.courses.find(x => x.id == cid);
        if(!c) { document.getElementById('scoreArea').innerHTML = ""; return; }
        
        let h = "<table><tr><th>Hole</th><th>Scr</th><th>Put</th><th>Tee</th></tr>";
        c.pars.forEach((p, i) => {
            const s = data ? data.rawScores[i] : "";
            const pt = data ? data.rawPutts[i] : 2;
            const tee = data ? data.rawTee[i] : "C";
            h += `<tr><td>${i+1}<br><small>P${p}</small></td>
            <td><input type="number" class="sv num-in" data-p="${p}" data-si="${c.sis[i]}" value="${s}"></td>
            <td><input type="number" class="pv num-in" value="${pt}"></td>
            <td><select class="tv" style="width:45px; padding:4px;" ${p==3?'disabled':''}>
                <option value="L" ${tee=='L'?'selected':''}>L</option>
                <option value="C" ${tee=='C'?'selected':''}>C</option>
                <option value="R" ${tee=='R'?'selected':''}>R</option>
            </select></td></tr>`;
        });
        document.getElementById('scoreArea').innerHTML = h + "</table>";
        document.getElementById('saveBtn').style.display = 'block';
    }

    function saveRound() {
        const pid = document.getElementById('pickPlayer').value;
        const cid = document.getElementById('pickCourse').value;
        if(!pid || !cid) return alert("Select Player and Course");
        
        const p = db.players.find(x => x.id == pid);
        const c = db.courses.find(x => x.id == cid);
        const editId = document.getElementById('editId').value;

        let pts=0, gross=0, putts=0, hp=0, rs=[], rp=[], rt=[];
        document.querySelectorAll('.sv').forEach((el, i) => {
            const v = parseInt(el.value);
            if(!isNaN(v) && v > 0) {
                hp++; rs.push(v);
                rp.push(parseInt(document.querySelectorAll('.pv')[i].value));
                rt.push(document.querySelectorAll('.tv')[i].value);
                let allow = (p.hcp >= el.dataset.si ? 1 : 0) + (p.hcp - 18 >= el.dataset.si ? 1 : 0);
                pts += Math.max(0, parseInt(el.dataset.p) - (v - allow) + 2);
                gross += v; putts += rp[rp.length-1];
            } else { rs.push(0); rp.push(0); rt.push(""); }
        });

        const diff = hp === 18 ? parseFloat(((113 / c.slope) * (gross - c.rating)).toFixed(1)) : null;
        const round = { id: editId ? parseInt(editId) : Date.now(), playerName: p.name, courseName: c.name, pts, gross, putts, diff, rawScores: rs, rawPutts: rp, rawTee: rt, date: document.getElementById('roundDate').value };

        if(editId) db.rounds[db.rounds.findIndex(r => r.id == editId)] = round;
        else db.rounds.push(round);
        
        sync();
        resetScore();
        document.querySelector('[data-target="pg-rounds"]').click();
    }

    function renderStats() {
        const f = document.getElementById('dashFilter').value;
        const pR = db.rounds.filter(r => r.playerName === f);
        const pRF = pR.filter(r => r.diff !== null);
        
        // WHS
        if(pRF.length >= 3) {
            const last20 = pRF.slice(-20);
            const count = last20.length < 6 ? 1 : last20.length < 12 ? 2 : 8;
            const best = last20.map(r => r.diff).sort((a,b)=>a-b).slice(0, count);
            document.getElementById('whsDisplay').innerText = (best.reduce((a,b)=>a+b,0)/best.length).toFixed(1);
        } else { document.getElementById('whsDisplay').innerText = "--"; }

        // Tendency
        let L=0, C=0, R=0, total=0;
        pR.forEach(rd => rd.rawTee.forEach(t => { if(t){ total++; if(t=='L') L++; else if(t=='C') C++; else if(t=='R') R++; } }));
        if(total > 0) {
            const lp=L/total*100, cp=C/total*100, rp=R/total*100;
            document.getElementById('tendencyBox').innerHTML = `<div class="t-bar"><div class="t-L" style="width:${lp}%"></div><div class="t-C" style="width:${cp}%"></div><div class="t-R" style="width:${rp}%"></div></div><p style="font-size:10px; text-align:center;">L:${lp.toFixed(0)}% | C:${cp.toFixed(0)}% | R:${rp.toFixed(0)}%</p>`;
        } else { document.getElementById('tendencyBox').innerHTML = ""; }

        document.getElementById('statsBox').innerHTML = pR.length ? `<div class="card" style="text-align:center;">Avg Pts<br><b>${(pR.reduce((a,b)=>a+b.pts,0)/pR.length).toFixed(1)}</b></div><div class="card" style="text-align:center;">Avg Putts<br><b>${(pR.reduce((a,b)=>a+b.putts,0)/pR.length).toFixed(1)}</b></div>` : "";
    }

    function renderRounds() {
        document.getElementById('resultsList').innerHTML = db.rounds.slice().reverse().map(r => `
            <div class="card">
                <div style="display:flex; justify-content:space-between;">
                    <div><b>${r.playerName}</b><br><small>${r.courseName}</small></div>
                    <div style="text-align:right"><b>${r.pts} Pts</b><br><span style="font-size:10px; background:var(--accent); color:white; padding:2px 5px; border-radius:4px;">${r.diff || 'PART'}</span></div>
                </div>
                <div style="margin-top:10px; display:flex; gap:5px;">
                    <button class="btn-sm" style="border:1px solid #ccc;" onclick="editRound(${r.id})">Edit</button>
                    <button class="btn-sm btn-danger" onclick="deleteR(${r.id})">Del</button>
                </div>
            </div>`).join('') || "No rounds.";
    }

    function editRound(id) {
        const r = db.rounds.find(x => x.id == id);
        document.getElementById('editId').value = id;
        document.getElementById('scoreTitle').innerText = "Edit Round";
        document.getElementById('roundDate').value = r.date;
        document.getElementById('pickPlayer').value = db.players.find(p => p.name === r.playerName).id;
        document.getElementById('pickCourse').value = db.courses.find(c => c.name === r.courseName).id;
        document.getElementById('scoreSelectors').style.display = 'none';
        document.getElementById('cancelBtn').style.display = 'block';
        buildScorecard(r);
        document.querySelector('[data-target="pg-score"]').click();
    }

    function resetScore() {
        document.getElementById('editId').value = "";
        document.getElementById('scoreTitle').innerText = "New Round";
        document.getElementById('scoreArea').innerHTML = "";
        document.getElementById('saveBtn').style.display = 'none';
        document.getElementById('cancelBtn').style.display = 'none';
        document.getElementById('scoreSelectors').style.display = 'block';
    }

    function deleteR(id) { if(confirm("Delete round?")) { db.rounds = db.rounds.filter(r => r.id !== id); sync(); renderRounds(); } }
    function deleteP(id) { if(confirm("Delete player?")) { db.players = db.players.filter(p => p.id !== id); sync(); } }
    function deleteC(id) { if(confirm("Delete course?")) { db.courses = db.courses.filter(c => c.id !== id); sync(); } }

    function updateDropdowns() {
        const p = document.getElementById('pickPlayer'), c = document.getElementById('pickCourse'), f = document.getElementById('dashFilter');
        p.innerHTML = '<option value="">-- Player --</option>' + db.players.map(x => `<option value="${x.id}">${x.name}</option>`).join('');
        c.innerHTML = '<option value="">-- Course --</option>' + db.courses.map(x => `<option value="${x.id}">${x.name}</option>`).join('');
        f.innerHTML = db.players.map(x => `<option value="${x.name}">${x.name}</option>`).join('');
    }

    function renderPlayerList() { document.getElementById('playerList').innerHTML = db.players.map(p => `<div style="display:flex; justify-content:space-between; padding:8px 0; border-bottom:1px solid #eee;">${p.name} (Hcp: ${p.hcp}) <button class="btn-sm btn-danger" onclick="deleteP(${p.id})">X</button></div>`).join(''); }
    function renderCourseList() { document.getElementById('courseList').innerHTML = db.courses.map(c => `<div style="display:flex; justify-content:space-between; padding:8px 0; border-bottom:1px solid #eee;">${c.name} <button class="btn-sm btn-danger" onclick="deleteC(${c.id})">X</button></div>`).join(''); }

    // --- STARTUP ---
    document.addEventListener("DOMContentLoaded", () => {
        const saved = localStorage.getItem(KEY);
        if(saved) db = JSON.parse(saved);
        
        // Build Course Setup
        let hLine = "<tr><th>Hole</th><th>Par</th><th>S.I.</th></tr>";
        for(let i=1; i<=18; i++) { hLine += `<tr><td>${i}</td><td><input type="number" class="par-in num-in" value="4"></td><td><input type="number" class="si-in num-in" value="${i}"></td></tr>`; }
        document.getElementById('setupTable').innerHTML = hLine;
        document.getElementById('roundDate').value = new Date().toISOString().split('T')[0];
        
        // Listeners
        document.getElementById('pickCourse').onchange = () => buildScorecard();
        document.getElementById('saveBtn').onclick = () => saveRound();
        document.getElementById('cancelBtn').onclick = () => resetScore();
        document.getElementById('addPlayerBtn').onclick = () => {
            const n = document.getElementById('pName').value.trim();
            const h = document.getElementById('pHcp').value || 0;
            if(n) { db.players.push({ id: Date.now(), name: n, hcp: parseFloat(h) }); document.getElementById('pName').value = ""; sync(); }
        };
        document.getElementById('addCourseBtn').onclick = () => {
            const n = document.getElementById('cName').value.trim();
            if(n) {
                const ps = Array.from(document.querySelectorAll('.par-in')).map(i => parseInt(i.value));
                const ss = Array.from(document.querySelectorAll('.si-in')).map(i => parseInt(i.value));
                db.courses.push({ id: Date.now(), name: n, rating: parseFloat(document.getElementById('cRate').value), slope: parseInt(document.getElementById('cSlope').value), pars: ps, sis: ss });
                document.getElementById('cName').value = ""; sync();
            }
        };

        initNav();
        sync();
        renderStats();
    });
</script>
